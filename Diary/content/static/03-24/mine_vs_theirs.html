Comparison of my models and theirs

% Setup code

if ~exist('RefModel', 'var') || ~exist('ref_model', 'var')

error('Unable to locate both reference models. Have you run both scripts ?');

end

bodeopts = bodeoptions('cstprefs');

bodeopts.Grid = 'on';

First off, the step responses

figure()

step(RefModel_dt, 'b', discreto, 'r');

hold on;

step(RefModel, 'b--', ref_model, 'r--');

legend('Mine', 'Theirs');

title('Step Reponses of both reference models');

ax = gca;

hold on;

plot(ax.XLim, [1.25, 1.25], 'g--');

plot(ax.XLim, [.75, .75], 'g--');

print -dpng 'mine_vs_theirs_step'

steps = [stepinfo(RefModel) stepinfo(ref_model)];

Next, the Bode Diagrams

figure()

bode(RefModel, ref_model, bodeopts);

legend('Mine', 'Theirs');

title('Bode Diagrams for both reference models');

print -dpng 'mine_vs_theirs_bode'
